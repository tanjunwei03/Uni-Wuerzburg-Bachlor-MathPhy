\begin{Problem}[Entanglement of a Pure 2-Qubit State]
	Consider the 2-qubit state
	\[
	\rho = |\psi\rangle\langle\psi|
	\]
	with
	\[
	|\psi\rangle
	=
	\frac15|00\rangle
	+
	\frac{2i\sqrt2}{5}|01\rangle
	-
	\frac{4}{15}|10\rangle
	-
	\frac{8i\sqrt2}{15}|11\rangle.
	\]
	
	\begin{enumerate}
		\item[(a)]
		Use \emph{Mathematica} or similar tools to compute the $4\times4$
		matrix $\rho$ in the qubit basis.
		Verify that $\rho$ is pure.
		
		\item[(b)]
		Compute the reduced density matrices
		$\rho^{(L)}$ and $\rho^{(R)}$
		for the left and right qubit by carrying out the partial traces
		\[
		\rho^{(L)}_{jk}
		=
		\sum_{m=0}^{1}
		\rho_{jm,km},
		\qquad
		\rho^{(R)}_{mn}
		=
		\sum_{j=0}^{1}
		\rho_{jm,jn}.
		\]
		
		\item[(c)]
		A pure state is called entangled if the reduced density matrices are
		mixed.
		Find out whether $\rho$ is entangled or not.
	\end{enumerate}
\end{Problem}
\begin{proof}
	\begin{parts}
	\item We define the ordered basis of the Hilbert space as $\{\ket{00}, \ket{01}, \ket{10}, \ket{11}\} $ (in that order). Thus, the ket would be represented as
		\[
		\ket{\psi} = \begin{pmatrix} \frac{1}{5}\\ \frac{2i\sqrt{2} }{5} \\ -\frac{4}{15}\\ -\frac{8i\sqrt{2} }{15} \end{pmatrix} 
		.\] 
		We then compute the outer product in mathematica using
		\begin{mmaCell}{Input}
v = {1/5, 2 I Sqrt[2]/5, -4/15, -8 I Sqrt[2]/15}
		\end{mmaCell}
		\begin{mmaCell}{Input}
			\mmaUnd{\(\rho\)} = Outer[Times, Conjugate[v], v]
		\end{mmaCell}
		\begin{mmaCell}{Output}
\{\{1/25, (2 I Sqrt[2])/25, -(4/75), -((8 I Sqrt[2])/75)\}, \{-((2 I Sqrt[2])/25), 8/25, (8 I Sqrt[2])/75, -(32/75)\}, \{-(4/75), -((8 I Sqrt[2])/75), 16/225, (32 I Sqrt[2])/225\}, \{(8 I Sqrt[2])/75, -(32/75), -((32 I Sqrt[2])/225), 128/225\}\}
		\end{mmaCell}
		to find
		\[
		\rho =\left(
\begin{array}{cccc}
 \frac{1}{25} & \frac{2 i \sqrt{2}}{25} & -\frac{4}{75} & -\frac{8 i \sqrt{2}}{75} \\
 -\frac{2 i \sqrt{2}}{25} & \frac{8}{25} & \frac{8 i \sqrt{2}}{75} & -\frac{32}{75} \\
 -\frac{4}{75} & -\frac{8 i \sqrt{2}}{75} & \frac{16}{225} & \frac{32 i \sqrt{2}}{225} \\
 \frac{8 i \sqrt{2}}{75} & -\frac{32}{75} & -\frac{32 i \sqrt{2}}{225} & \frac{128}{225} \\
\end{array}
\right) 
		.\] 
		We can square this matrix to get
		\[
		\rho^2 =\left(
\begin{array}{cccc}
 \frac{1}{25} & \frac{2 i \sqrt{2}}{25} & -\frac{4}{75} & -\frac{8 i \sqrt{2}}{75} \\
 -\frac{2 i \sqrt{2}}{25} & \frac{8}{25} & \frac{8 i \sqrt{2}}{75} & -\frac{32}{75} \\
 -\frac{4}{75} & -\frac{8 i \sqrt{2}}{75} & \frac{16}{225} & \frac{32 i \sqrt{2}}{225} \\
 \frac{8 i \sqrt{2}}{75} & -\frac{32}{75} & -\frac{32 i \sqrt{2}}{225} & \frac{128}{225} \\
\end{array}
\right) 
		.\]
		Since the matrix has not changed at all, we know that the trace remains the same, i.e. it is 1. We can also confirm this by evaluating
		\begin{mmaCell}{Input}
			Tr[\mmaUnd{\(\rho\)}.\mmaUnd{\(\rho\)}]
		\end{mmaCell}
		\begin{mmaCell}{Output}
			1
		\end{mmaCell}
		This is to be expected, since we had $\rho = \dyad{\psi}{\psi}$ and hence
		\[
			\rho^2 = \ket{\psi}\braket{\psi}{\psi}\bra{\psi}=\dyad{\psi}
		.\] 
	\item In terms of our indexing, the multiindex $(m, j)$ is mapped to the index $2(m-1)+j$, where we note that the indices $m$ and $j$ take values $\{0,1\} $.

		Thus, we can rewrite the reduced matrices as
		\begin{align*}
			\rho_{jk}^{(L)} &= \sum_{m=0}^1 \rho_{jm,km}\\
			&= \sum_{m=0}^1 \rho_{(2(j-1)+m)(2(k-1)+m)}
		\end{align*}
		We can evaluate this with mathematica as
		\begin{mmaCell}{Input}
Table[Sum[\mmaUnd{\(\rho\)}[[2 (j - 1) + m, 2 (k - 1) + m]], \{m, 1, 2\}], \{j, 1, 2\}, \{k, 1, 2\}]
		\end{mmaCell}
		\begin{mmaCell}{Output}
\{\{9/25, -(12/25)\}, \{-(12/25), 16/25\}\}
		\end{mmaCell}
		We can also evaluate the right reduced density matrix as
		\begin{align*}
			\rho_{mn}^{(R)} &= \sum_{j=1}^2 \rho_{jm,jn} \\
					&= \sum_{j=1}^2 \rho_{(2(j-1)+m)(2(j-1)+n)} 
		\end{align*}
		We then evaluate this using
		\begin{mmaCell}{Input}
Table[Sum[\mmaUnd{\(\rho\)}[[2 (j - 1) + m, 2 (j - 1) + n]], \{j, 1, 2\}], \{m, 1, 2\}, \{n, 1, 2\}]
		\end{mmaCell}
		\begin{mmaCell}{Output}
\{\{1/9, (2 I Sqrt[2])/9\}, \{-((2 I Sqrt[2])/9), 8/9\}\}
		\end{mmaCell}
		In a better formatted matrix form, we have
		\[
			\rho^{(L)} =\left(
\begin{array}{cc}
 \frac{9}{25} & -\frac{12}{25} \\
 -\frac{12}{25} & \frac{16}{25} \\
\end{array}
\right), \qquad \rho^{(R)} =\left(
\begin{array}{cc}
 \frac{1}{9} & \frac{2 i \sqrt{2}}{9} \\
 -\frac{2 i \sqrt{2}}{9} & \frac{8}{9} \\
\end{array}
\right)  
		.\] 
	\item We can verify
		\[
			(\rho^{(L)})^2 = \rho_L,\qquad (\rho^{(R)})^2 = \rho^{(R)}
		\]
		using the following code
		\begin{mmaCell}{Input}
\mmaUnd{\(\rho\)L}.\mmaUnd{\(\rho\)L}
		\end{mmaCell}
		\begin{mmaCell}{Output}
\{\{9/25, -(12/25)\}, \{-(12/25), 16/25\}\}
		\end{mmaCell}
		\begin{mmaCell}{Input}
\mmaUnd{\(\rho\)R}.\mmaUnd{\(\rho\)R}
		\end{mmaCell}
		\begin{mmaCell}{Output}
\{\{1/9, (2 I Sqrt[2])/9\}, \{-((2 I Sqrt[2])/9), 8/9\}\}
		\end{mmaCell}
		Since $\tr(\rho^{(L)})=\tr(\rho^{(R)})=1$, we have $\tr((\rho^{(L)})^2)=\tr((\rho^{(R)})^2)=1$, i.e. the states are pure. This means that the states are not entangled.
	\end{parts}
\end{proof}
\begin{Problem}[Quantum Cryptography]
	
	Encrypted communication requires that both parties share a key (a random
	sequence of classical bits) that can be used to encode and decode messages
	by bitwise XOR operations.
	The problem of encryption therefore reduces to safely generating identical
	keys on both sides.
	
	Alice randomly generates a classical bit
	$a\in\{0,1\}$,
	using it to prepare a qubit
	
	\[
	|\psi\rangle
	=
	\begin{cases}
		|0\rangle,&a=0,\\
		|+\rangle,&a=1,
	\end{cases}
	\]
	
	where
	
	\[
	|\pm\rangle
	=
	\frac1{\sqrt2}(|0\rangle\pm|1\rangle).
	\]
	
	She sends this qubit to Bob.
	
	Likewise Bob randomly generates a classical bit
	$b\in\{0,1\}$.
	He chooses one of the two measurements
	
	\[
	\mathbf{M}_0
	=
	|0\rangle\langle0|
	-
	|1\rangle\langle1|,
	\qquad
	\mathbf{M}_1
	=
	|+\rangle\langle+|
	-
	|-\rangle\langle-|.
	\]
	
	Bob measures the incoming qubit and communicates the measurement result
	$m=\pm1$ publicly.
	They repeat the protocol until $m=-1$.
	
	\begin{enumerate}
		\item[(a)]
		Show that
		$a=1-b$
		if $m=-1$.
		
		\item[(b)]
		How many qubits are on average needed to generate one bit of the key?
		
		\item[(c)]
	Suppose that Eve intercepts the communication in that she measures the transferred qubit by a projective measurement $\mathbf{E}=|\phi_+\rangle\langle\phi_+|
		-
		|\phi_-\rangle\langle\phi_-|$, where
		\[
		|\phi_+\rangle
		=
		\alpha|0\rangle+\beta|1\rangle,\qquad
		|\phi_-\rangle
		=
		\beta^*|0\rangle-\alpha^*|1\rangle,\qquad
		|\alpha|^2+|\beta|^2=1.
		\]
are two arbitrary orthogonal states. Show that Eve’s interception generally changes the measurement statistics seen from the perspective of Bob and Alice, depending effectively on two parameters  
$
		u:=|\alpha|^2$ and $v:=\frac12|\alpha+\beta|^2$.
		
		\item[(d)]
		Determine $u$ and $v$ such that Bob and Alice cannot detect the
		interception.
		
		\item[(e)]
		In the case of part (d), is it possible for Eve to obtain information
		about the key?
		
	\end{enumerate}
\end{Problem}
\begin{proof}
	\begin{parts}
	\item Suppose $a=0$. Thus Alice sends the state $\ket{0}$ to Bob. If Bob chooses $M_0$, the chance of measuring $m=-1$ is given by
		\[
			\mathbbm{P}(m=-1|M_0,\ket{0})=|\braket{1}{0}|^2=0
		.\] 
		Thus, if Bob chooses $M_0$, there is no chance of measuring -1. This means that Bob must have chosen $M_1$, i.e. $b=1$.

		Now suppose $a=1$, and Alice sends the state $\ket{1}$ to Bob. If Bob has chosen the apparatus $M_1$, the probability of a measurement yielding -1 is given by
		\[
			\mathbbm{P}(m=-1|M_1,\ket{+})=|\braket{-}{+}|^2=0
		.\] 
		Thus, if Bob has measured a -1, he must have chosen the apparatus $M_0$. We are then forced to conclude $b=0$.

		In any case, $a=1-b$.

		For the sake of completeness, we compute the probability of measuring $m=-1$ when using the ``correct'' measurement apparatus:
		\begin{align*}
			\mathbbm{P}(m=-1|M_1,\ket{0})&=|\braket{-}{0}|^2 = \frac{1}{2} \\
			\mathbbm{P}(m=-1|M_0,\ket{+})&= |\braket{1}{+}|^2 = \frac{1}{2} 
		\end{align*}
	\item As we proved in the previous part, the probability of measuring $m=-1$ is $\frac{1}{2}$ given that we use the ``correct'' apparatus (and $0$ otherwise). Since the probability that we use the ``correct'' apparatus is $\frac{1}{2}$, we have, independently of the value of $a$,
		\[
			\mathbbm{P}(m=-1)=\mathbbm{P}(\text{correct apparatus})\cdot \frac{1}{2}= \frac{1}{4}
		.\] 
		We thus need 4 qubits on average to generate one bit of the key.

			\braket{1}{\phi_+} &= \beta, &|\braket{1}{\phi_+}|^2 &= |\beta|^2=1-u \\
			\braket{0}{\phi_-} &= \beta^*, &|\braket{0}{\phi_-}|^2 &= |\beta|^2 = 1-u\\
			\braket{1}{\phi_-} &= -\alpha^*, & |\braket{1}{\phi_-}|^2 &= |\alpha|^2 = u\\
			\braket{-}{\phi_+} &= \frac{1}{\sqrt{2} }(\alpha - \beta), &|\braket{-}{\phi_+}|^2 &= \frac{1}{2}|\alpha - \beta|^2 = 1-v  \\
			\braket{+}{\phi_+} &= \frac{1}{\sqrt{2} }(\alpha + \beta) \\
			\braket{-}{\phi_-} &= \frac{1}{\sqrt{2} }(\beta^* + \alpha^*) \\
			\braket{+}{\phi_-} &= \frac{1}{\sqrt{2} }(\beta^* - \alpha^*) 
		\end{align*}
