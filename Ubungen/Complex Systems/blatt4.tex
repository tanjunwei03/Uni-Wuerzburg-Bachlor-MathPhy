\begin{Problem}[Convolution product (6P)]
	
	Let us consider the probability density of the gaps between Poisson-distributed events
	(the length $x$ of silence between the clicks of a Geiger counter), which is defined by
	\[
	p(x)=w e^{-wx}
	\qquad (x\ge 0,\; w>0).
	\]
	
	\begin{enumerate}[label=(\alph*)]
		\item Compute the corresponding characteristic function (see lecture notes). \hfill (1P)
		
		\item Compute the two-fold convolution product $(p*p)(x)$ and the three-fold convolution
		product $(p*p*p)(x)$. \hfill (2P)
		
		\item Guess a general formula for the $N$-fold convolution product
		$p^{*N}(x)$ and prove it by recursion. \hfill (2P)
		
		\item Fit a normal distribution with mean $\mu$ and width $\sigma$ to the probability
		density computed in (c) for large $N$.
		
		\textit{Hint:} Recall the behavior of cumulants. \hfill (1P)
	\end{enumerate}
\end{Problem}
\begin{proof}
	\begin{parts}
	\item The characteristic function is given by
		\[
			\phi_X(t) = E[e^{itX}]
		.\] 
		We simply compute this
		\begin{align*}
			\phi_X(t) &= E[e^{itX}] \\
				  &= \int_0^\infty e^{itx}we^{-wx}\dd{x} \\
				  &= w \int_0^\infty e^{(it-w)x}\dd{x} \\
				  &= w\left[ \frac{1}{it-w}e^{(it-w)x} \right]_0^\infty  \\
				  &= \frac{w}{w-it}
		\end{align*}
	\item The convolution of two probability distributions is given by
		\[
			(f*g)(z)=\int_{-\infty}^\infty f(z-t)g(t)\dd{t}
		.\] 
		Again, we substitute this in, using a step function $\theta(t)$ to ensure that the probability distribution of the Poisson distribution only has support on the positive real axis.
		\begin{align*}
			(p*p)(z) &= \int_{-\infty}^\infty we^{-w(z-t)}\theta(z-t)we^{-wt}\theta(t) \dd{t}\\
				 &= w^2 \theta(z)\int_0^z e^{-wz}e^{wt}e^{-wt}\dd{t} \\
				 &= w^2 ze^{-wz} \theta(z)
		\end{align*}
		We compute the threefold product similarly
		\begin{align*}
			(p*p*p)(z) &= w^3 \int_{-\infty}^\infty e^{-w(z-t)}\theta(z-t)te^{-wt}\theta(t)\dd{t} \\
				   &= w^3 \theta(z)\int_0^z e^{-wz}t\dd{t}\\
				   &= w^3 \frac{z^2}{2}e^{-wz} 
		\end{align*}
	\item We conjecture that
		\[
			p^{*N}(x) = w^N \frac{x^{N-1}}{(N-1)!}e^{-wx}\theta(x) 
		.\] 
		We prove this by induction. Suppose this is true for $N$. We will prove it is true for $N+ 1$. We compute this product directly
		\begin{align*}
			p^{*(N+1)}(z) &= (p*p^{*N})(z) \\
				      &= \int_{-\infty}^\infty we^{-w(z-t)}\theta(z-t)w^N \frac{t^{N-1}}{(N-1)!}e^{-wt}\theta(t)\dd{t} \\
				      &= w^{N+1} \theta(z)\int_0^z e^{-wz} \frac{t^{N-1}}{(N-1)!}\dd{t} \\
				      &= w^{N+1} \frac{t^{N}}{N!}e^{-wz}\theta(z)
		\end{align*}
		Since the base cases with $N=1$ and $N=2$ have already been proven, this proves the theorem.
	\item By a suitable analytic continuation, we compute the moment generating function
		\[
		M_X(t) = \phi_X(-it)=\frac{w}{w-t}
		.\] 
		Then the cumulant generating function is
		\begin{align*}
			K_X(t) &= \ln M_X(t) = \ln \frac{w}{w-t}=-\ln \left( 1-\frac{t}{w} \right)\\
			&= \sum_{n=1}^{\infty} \frac{1}{n}\left( \frac{t}{w} \right)^n  
	\end{align*}
We know that the cumulant generating function of the sum of $N$ random variables is the sum of the CGFs. Thus, the CGF of $p^{*N}$ is given by
\[K_{NX}(t) = -N\ln \left( 1-\frac{t}{w} \right) =\sum_{n=1}^{\infty} \frac{N}{nw^n}t^n\]
We first standardise the distribution; we do this for only one variable, since both the variance and expectation values are linear. We have the expectation value
\begin{align*}
	\mathbb{E}[X] &= \int_0^\infty xwe^{-wx}\dd{x} = \frac{1}{w} \\
	\text{Var}(X) &= \mathbb{E}[X^2]-\mathbb{E}[X]^2 \\
	\mathbb{E}[X^2] &= \int_0^\infty x^2 we^{-wx}\dd{x} = \frac{2}{w^2} \\
	\text{Var}(X) &= \frac{1}{w^2} 
\end{align*}
Thus, we standardise the distribution as
\[
Z_N = \frac{X_N - \mu_N}{\sigma_N}
\]
where $\mu_N = N / w$ and $\sigma_N = \sqrt{N} / w$. The cumulant generating function of $Z_N$ is given by
\begin{align*}
	K_{Z_N} &= - \frac{\mu_N}{\sigma_N}t + K_{NX}\left( \frac{t}{\sigma_N} \right)  \\
	&= -\sqrt{N} t- N \ln \left( 1 - \frac{t}{\sqrt{N} } \right)  \\
	&= -\sqrt{N} t + N \sum_{n=1}^{\infty} \frac{1}{n}\left( \frac{t}{\sqrt{N} } \right)^n \\
	&= N \sum_{n=2}^{\infty} \frac{1}{n}\left( \frac{t}{\sqrt{N} } \right)^n \\
	&=  \sum_{n=2}^{\infty} \frac{1}{n}N^{1 - n / 2} t^n
\end{align*}
Again, we find that the only cumulant that does not vanish at $N\to\infty$ is the $N^0=\kappa_2$ second cumulant. Thus, the standardised distribution converges to the CGF of a standard normal distribution. Inverting, we find that
\[
	p^{*N} \xrightarrow{N\to\infty} \mathcal{N}\left( \frac{N}{w}, \frac{N}{w^2} \right) 
\] 
where $\mathcal{N}(\mu,\sigma^2)$ represents the probability density function of a normal distribution with mean $\mu$ and variance $\sigma^2$. Explicitly, we have
\[
	p^{*N}(x) \xrightarrow{N\to\infty} \frac{w}{\sqrt{2\pi N} } e^{-\frac{(N-wx)^2}{2N}} 
.\] 
In other words, we have a mean $\mu = \frac{N}{w}$ and a standard deviation $\sigma = \frac{\sqrt{N}}{w}$. This is a relatively good agreement; see Fig. \ref{fig:complexsystemsblatt4-1}.\qedhere

\begin{Problem}[Harmonic oscillator (6P)]
	
	Consider a harmonic oscillator with eigenstates
	\[
	|n\rangle = |0\rangle, |1\rangle, |2\rangle, \ldots
	\]
	and eigenenergies
	\[
	E_n = \hbar \omega \left(n+\frac12\right).
	\]
	
	Let us assume that the eigenstates correspond to classical configurations and that
	the system evolves stochastically according to a continuous-time Markov process as follows:
	\[
	n \to n+1 \qquad \text{with rate } \omega_\uparrow
	\]
	\[
	n \to n-1 \qquad \text{with rate } \omega_\downarrow
	\]
	
	\begin{enumerate}[label=(\alph*)]
		\item Calculate the Liouville operator $\mathcal{L}$. \hfill (1P)
		
		\item Determine the left and the right eigenvector to the eigenvalue zero. \hfill (2P)
		
		\item Normalize the right eigenvector obtained in (b) so that it can be interpreted
		as a probability distribution. Under which condition is the normalization possible?
		\hfill (2P)
		
		\item In a canonical ensemble the probability of finding the system at energy $E_n$
		is proportional to
		\[
		\exp\!\left(-\frac{E_n}{k_B T}\right),
		\]
		where $T$ is the temperature and $k_B$ is the Boltzmann constant.
		Express the temperature $T$ in terms of the rates
		$\omega_\uparrow$ and $\omega_\downarrow$. \hfill (1P)
	\end{enumerate}
\end{Problem}
\begin{proof}
	\begin{parts}
	\item Recall that the Liouvillian is defined through the master equation
		\[
			\partial_t \ket{\Psi} = -\mathcal{L}\ket{\Psi}
		.\] 
		Since it is stated that we may assume the eigenstates correspond to classical configurations, there is no need to concern ourselves with possible superpositions. The transition rates, by definition, lead to the following equations for the time evolution of the occupation probabilities
		\[
			\pdv{P_n}{t} = -(\omega_\downarrow+\omega_\uparrow)P_n + \omega_\downarrow P_{n+1} + \omega_\uparrow P_{n-1}
		.\] 
		Note that this equation does not work for $n=0$, since obviously there is no state at $-1$. We will handle this separately, but the general argument remains. Since each individual probability is the component of a vector $P_n = \braket{n}{\Psi}$, we find that
		\begin{align*}
			\pdv{t}\braket{n}{\Psi} &= -(\omega_\downarrow+\omega_\uparrow)\braket{n}{P_t} + \omega_\downarrow \braket{n+1}{\Psi} + \omega_\uparrow \braket{n-1}{\Psi}\\
			\mel{n}{\pdv{t}}{\Psi} &= \left[-(\omega_\downarrow+\omega_\uparrow)\bra{n} + \omega_\downarrow \bra{n+1} + \omega_\uparrow \bra{n-1}\right]\ket{\psi}\\
			\pdv{t}\ket{\Psi} &= \sum_n\ket{n}\left[-(\omega_\downarrow+\omega_\uparrow)\bra{n} + \omega_\downarrow \bra{n+1} + \omega_\uparrow \bra{n-1}\right]\ket{\psi}
		\end{align*}
		Handling the $n=0$ case properly and writing the sum bounds explicitly, we find that the Liouvillian operator is given by
		\[
			\mathcal{L}=-\omega_{\downarrow}\dyad{0}{1}+\omega_{\uparrow}\dyad{0}{0} - \sum_{n=1}^{\infty} \left[ \omega_{\uparrow}\dyad{n}{n-1} + \omega_{\downarrow}\dyad{n}{n+1} -(\omega_{\uparrow}+\omega_{\downarrow})\dyad{n}{n}\right] 
		.\] 
	\item For the left eigenvector, we require that 
		\begin{align*}
			0 =&~\bra{\Psi_L}\mathcal{L} \\
			=&~-\omega_\downarrow\braket{\Psi_L}{0}\bra{1}+\omega_\uparrow \braket{\Psi_L}{0}\bra{0}\\
			 &-\sum_{n=1}^{\infty}\left[ \omega_\uparrow \braket{\Psi_L}{n}\bra{n-1}+\omega_\downarrow \braket{\Psi_L}{n}\bra{n+1}-(\omega_\uparrow+\omega_\downarrow)\braket{\Psi_L}{n}\bra{n}  \right]\\
			=&~-\omega_\downarrow\braket{\Psi_L}{0}\bra{1}+\omega_\uparrow \braket{\Psi_L}{0}\bra{0}\\
			 &-\sum_{n=0}^{\infty} \omega_\uparrow \braket{\Psi_L}{n+1}\bra{n}-\sum_{n=2}^{\infty} \omega_\downarrow \braket{\Psi_L}{n-1}\bra{n}+\sum_{n=1}^{\infty} (\omega_\uparrow+\omega_\downarrow)\braket{\Psi_L}{n}\bra{n}  \\
			=&~\omega_\uparrow \braket{\Psi_L}{0}\bra{0}-\omega_\uparrow \braket{\Psi_L}{1}\bra{0}\\
			 &-\sum_{n=1}^{\infty} \omega_\uparrow \braket{\Psi_L}{n+1}\bra{n}-\sum_{n=1}^{\infty} \omega_\downarrow \braket{\Psi_L}{n-1}\bra{n}+\sum_{n=1}^{\infty} (\omega_\uparrow+\omega_\downarrow)\braket{\Psi_L}{n}\bra{n}  
		\end{align*}
		upon which we may compare coefficients of all the basis 1-forms $\bra{n}$. Comparing that of $\bra{0}$, we find that $\braket{\Psi_L}{0}=\braket{\Psi_L}{1}$. Then, for all $n\ge 1$, we have 
		\[
			(\omega_\uparrow+\omega_\downarrow)\braket{\Psi_L}{n} = \omega_\uparrow \braket{\Psi_L}{n+1} + \omega_\downarrow \braket{\Psi_L}{n-1}
		.\] 
		We now prove another statement by induction.
		\begin{tcolorbox}
			\begin{Lemma}
				If $\braket{\Psi_L}{n}=\braket{\Psi_L}{n-1}$, then $\braket{\Psi_L}{n}=\braket{\Psi_L}{n+1}$.
			\end{Lemma}
			\begin{proof}
				We have
				\begin{align*}
					(\omega_\uparrow+\omega_\downarrow)\braket{\Psi_L}{n} &= \omega_\uparrow \braket{\Psi_L}{n+1} + \omega_\downarrow \braket{\Psi_L}{n-1}\\
					(\omega_\uparrow+\omega_\downarrow)\braket{\Psi_L}{n} &= \omega_\uparrow \braket{\Psi_L}{n+1} + \omega_\downarrow \braket{\Psi_L}{n}\\
					\omega_\uparrow\braket{\Psi_L}{n} &= \omega_\uparrow \braket{\Psi_L}{n+1}\\ 
					\braket{\Psi_L}{n} &= \braket{\Psi_L}{n+1}.\qedhere 
				\end{align*}
			\end{proof}
		\end{tcolorbox}
		From this statement, since we know that $\braket{\Psi_L}{0}=\braket{\Psi_L}{1}$, we find that the vector has all components be identical, which is what we expect from a normalised probability distribution. Alternatively, we can write the left eigenvector as
		\[\Psi_L= \sum_{n=0}^\infty \bra{n}\]

		Now we compute the right eigenvector in a similar manner:
		\begin{align*}
			0 =& \mathcal{L}\ket{\Psi_R} \\
			=&~-\omega_\downarrow\ket{0}\braket{1}{\Psi_R} + \omega_\uparrow \ket{0}\braket{0}{\Psi_R}\\
			 &-\sum_{n=1}^{\infty} \left[ \omega_\uparrow\ket{n}\braket{n-1}{\Psi_R}+\omega_\downarrow \ket{n}\braket{n+1}{\Psi_R}-(\omega_\downarrow+\omega_\uparrow)\ket{n}\braket{n}{\Psi_R} \right] 
		\end{align*}
		Now, we again compare the coefficients of the ket vectors $\ket{n}$. By analysing the coefficient of $\ket{0}$, we find that
		\[
			\omega_\downarrow \braket{1}{\Psi_R} = \omega_\uparrow \braket{0}{\Psi_R}
		\] 
		or
		\[
			\braket{1}{\Psi_R} = \left( \frac{\omega_\uparrow}{\omega_\downarrow} \right) \braket{0}{\Psi_R}
		.\] 
		For $n\ge 1$, comparing coefficients of the $\ket{n}$ yields
		\[
			\omega_\uparrow \braket{n-1}{\Psi_R} +\omega_\downarrow \braket{n+1}{\Psi_R}-(\omega_\downarrow + \omega_\uparrow)\braket{n}{\Psi_R}=0
		.\] 
		We proceed again by induction. We first hypothesize that
		\[
			\braket{n}{\Psi_R} = \left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n \braket{0}{\Psi_R} 
		.\] 
		We know this is true for $n=1$ and $n=0$. Now, fix $n$, and assume the statement is true for all $k$ with $k\le n$ (strong induction). We will prove it is true for $n+1$. 
		\begin{align*}
			0&=\omega_\uparrow \braket{n-1}{\Psi_R} +\omega_\downarrow \braket{n+1}{\Psi_R}-(\omega_\downarrow + \omega_\uparrow)\braket{n}{\Psi_R}\\
			0&=\omega_\uparrow \left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^{n-1}\braket{\Psi_R}{0} +\omega_\downarrow \braket{n+1}{\Psi_R}-(\omega_\downarrow + \omega_\uparrow)\left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n \braket{\Psi_R}{0} \\
0&=\omega_\downarrow \braket{n+1}{\Psi_R}- \omega_\uparrow\left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n \braket{\Psi_R}{0} \\
\braket{n+1}{\Psi_R}&= \left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^{n+1} \braket{\Psi_R}{0}
		\end{align*}
		This proves the statement, and thus we can write the right eigenvector as
		\[\Psi_R = \sum_{n=0}^\infty\left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n\ket{n} \]
	\item We must first calculate the sum of all components of the vector; the normalisation is possible if and only if this sum is finite. We sum
		\begin{align*}
			\sum_{n=0}^{\infty} \braket{n}{\Psi_R} &= \sum_{n=0}^{\infty} \left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n \braket{0}{\Psi_R} \\
							     &= \frac{1}{1- \frac{\omega_\uparrow}{\omega_\downarrow}}\braket{0}{\Psi_R} & \left| \frac{\omega_\uparrow}{\omega_\downarrow} \right|<1 
		\end{align*}
		Dividing, we find that the components are given by
		\[
			\braket{n}{\Psi_R} = \left( 1- \frac{\omega_\uparrow}{\omega_\downarrow} \right)\left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n
		\]
		leading to a normalised vector
		\[\Psi_R = \sum_{n=0}^\infty \left( 1- \frac{\omega_\uparrow}{\omega_\downarrow} \right)\left( \frac{\omega_\uparrow}{\omega_\downarrow} \right)^n\ket{n}\]
		and this normalisation is possible if and only if we have
		\[
		\left| \frac{\omega_\uparrow}{\omega_\downarrow} \right| <1
		.\] 
	\item Noting the special form of the probability density as being a geometric progression, we seek to massage the probability density of the canonical ensemble into this form:
		\begin{align*}
			P_n &= \exp\left( - \frac{E_n}{k_B T} \right)  \\
			    &= \exp\left[ -\frac{\hbar\omega}{k_B T}\left( n+\frac{1}{2} \right)  \right]  \\
			    &= \exp\left(- \frac{\hbar\omega}{2k_B T} \right) \exp\left( -\frac{\hbar\omega n}{k_B T}  \right)  \\
			    &= \exp\left(- \frac{\hbar\omega}{2k_B T} \right) \exp\left( -\frac{\hbar\omega }{k_B T}  \right)^n  
		\end{align*}
		At this point, it is only necessary to compare the decay rate with $n$. Essentially, we would be comparing the ratio $P_{n+1} / P_n$. By comparing this ratio between the two expressions and demanding that it be the same, we find that
		\[
		\frac{\omega_\uparrow}{\omega_\downarrow}= \exp\left( -\frac{\hbar\omega}{k_B T} \right) 
		.\] 
		After some massaging, we find that
		\begin{align*}
			\ln \frac{\omega_\downarrow}{\omega_\uparrow}&= \frac{\hbar\omega}{k_B T}\\	
T &= \frac{\hbar\omega}{k_B \ln \frac{\omega_\downarrow}{\omega_\uparrow}}.\qedhere	
		\end{align*}
	\end{parts}
\end{proof}
